var

การสร้างตัวเเปรใน javascript สามารถทำใด้โดยการใช้คำสั่ง var
example

var num = 0 , name = 'jame';

จากตัวอย่างจะเห็นใด้ว่าใน javascript การประกาศตัวแปรจะไม่ใช้ชนิดของตัวเเปรมาใช้เพื่อประกาศตัวเเปร(เช่น int float string...) เเต่จะเป็นการประกาศเพียงช่อตัวเเปรเเล้วชนิดของตัวเเปรจะขึ้นอยู่กับตัวเเปรที่เก็บไว้

if&else

ใน javascript การใช้ if&else นั้นจะเหมื่อนกับภาษาอื่นๆโดยจะใช้ syntax ดังต่อไปนี้

if(condition_1){
        
    do when the condition_1 was true ;
        
} else if(condition_2){
        
    do when the condition_2 was true and condition_1 was false ;
        
} else {
        
    do when the condition was none above ;
        
}

switch case

การทำงานของ switch case จะเหมือนกับการใช้ if else ไปทีละ condition เเต่ switch case จะดูสั้นกว่าการใช้ if else โดย swith case จะมี syntax ดังต่อไปนี้

switch (expression) {
        
    case condition_1: do when condition_1 was true;
        
    break;
        
    
        
    case condition_2: do when condition_2 was true;
        
    break;
        
    ...
        
    
        
    case condition_n: do when condition_n was true;
        
    break;
        
    
        
    default: do when none of all above condition was true;
        
}

for loop

การใช้งาน for loop จะใช้เมื่อต้องการที่จะทำบางอย่างซ้ำๆไปเรื่อยๆจนกว่าจะใด้ผลที่ต้องการโดย for loop จะใช้ syntax ต่อไปนี้

for (varlueble ; condition ; iteration){
        
    statement;
        
}

example

var n = 0 ;
        
for(var i = 0 ; i < 20 ; i++){
        
    n += i ;
        
}

การใช้ for loop เพื่อนำค่าภายใน list มาใช้นั้นจะต้องใช้คำสั่ง

for ( var i = 0 ; i < some_list.length ; i++){
        
    statement for some_list[i];
        
}

ซื่งดูยุ่งยากเเต่ใน javascript นั้นมี for in มาให้ใช้เพื่อลดจำนวนตัวเเปลลงไปเเละง่ายต่อการอ่านโดยใช้ syntax ต่อไปนี้

for (varlueble in some_list){
        
    statement for varlueble;
        
}

while loop

การทำงานของ while loop นั้นเหมื่อนกับ for loop เพียงเเต่ต้องการเพียง condition เดียวโดยมี syntax ต่อไปนี้

while (condition){
        
     statement to excecute while condition was true;
        
}

do while จะเป็นการทำช้ำเหมื่อนกับ while loop เเต่จะต่างกันตรงที่เมื่อเจอ while เเล้ว ตัว condition เป็น true เเลัวจึงจะกลับไปทำที่ do

example

do {
        
    statement to excecute while condition was true;
        
} while (condition);
